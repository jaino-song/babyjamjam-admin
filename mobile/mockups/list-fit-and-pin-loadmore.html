// useLayoutEffect, 매 resize · resetKey 변경 시 재측정
const scroll = scrollContainerRef.current;
const firstRow = scroll.querySelector('.list-item, .contract-item');
const scrollH = scroll.clientHeight; // 이미 footer 제외됨 (flex sibling)
const rowH = firstRow.offsetHeight;
const gap = parseFloat(getComputedStyle(scroll).rowGap) || 8;
const fit = Math.max(1, Math.floor((scrollH + gap) / (rowH + gap)));
setVisibleCount(fit);
