// Intent detection function
export function detectIntent(input: string): {
  intent: string;
  tool: string;
  args?: Record<string, unknown>;
  requiresDisambiguation?: boolean;
} | null {
  const normalized = input.trim();

  // Sort by priority (highest first)
  const sortedPatterns = [...INTENT_PATTERNS].sort(
    (a, b) => b.priority - a.priority
  );

  for (const pattern of sortedPatterns) {
    const matched = pattern.patterns.some(p => p.test(normalized));

    if (matched) {
      if (pattern.disambiguationRequired) {
        return {
          intent: pattern.name,
          tool: pattern.tool,
          requiresDisambiguation: true
        };
      }

      let tool = pattern.tool;

      // Resolve entity placeholder
      if (pattern.requiresEntity) {
        const entity = detectEntityFromInput(normalized);
        if (entity) {
          tool = tool.replace('{{Entity}}', entity);
        } else {
          continue; // Skip if entity required but not found
        }
      }

      return {
        intent: pattern.name,
        tool,
        args: getIntentArgs(pattern.name)
      };
    }
  }

  return null;
}

function getIntentArgs(intentName: string): Record<string, unknown> | undefined {
  const argsMap: Record<string, Record<string, unknown>> = {
    endingSoon: { filter: 'ending-soon' },
    startingSoon: { filter: 'starting-soon' },
    noContract: { filter: 'no-contract' },
    incompleteContract: { filter: 'incomplete-contracts' }
  };
  return argsMap[intentName];
}

// response-format.ts

export function selectResponseFormat(resultCount: number): 'card' | 'list' | 'table' {
  if (resultCount === 1) return 'card';
  if (resultCount <= 5) return 'list';
  return 'table';
}

export const RESPONSE_TEMPLATES = {
  // Card format (single result)
  card: {
    client: (data: Client) => `## ${data.name} 산모님
📞 ${data.phone}
📍 ${data.address}
📅 서비스 기간: ${data.startDate} ~ ${data.endDate}
👩‍⚕️ 담당: ${data.assignedEmployee?.name ?? '미배정'}`,

    employee: (data: Employee) => `## ${data.name} ${data.grade}급
📞 ${data.phone}
📍 근무 가능 지역: ${data.workAreas.join(', ')}
📋 현재 상태: ${data.isAvailable ? '배정 가능' : '배정 중'}`
  },

  // List format (2-5 results)
  list: {
    client: (items: Client[]) => items
      .map(c => `- ${c.name} (${c.phone}) - ${c.status}`)
      .join('\n'),

    employee: (items: Employee[]) => items
      .map(e => `- ${e.name} ${e.grade}급 (${e.phone}) - ${e.isAvailable ? '가능' : '배정중'}`)
      .join('\n')
  },

  // Table format (6+ results)

    client: (items: Client[]) => {
      let table = `| 이름 | 연락처 | 상태 | 담당 관리사 |\n`;
